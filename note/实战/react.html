# 配置图片、字体的loader

（项目中图片格式推荐webp格式，因为体积更小）

module.exports={
    module:{
        rules:[
            {
                test: /\.(png|jpeg|gif|eot|woff|woff2|ttf|svg|otf|webp)$/,
                type:"asset"//内置了file-loader、url-loader
            }
        ]
    }
}
